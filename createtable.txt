CREATE TABLE project.teachers (
    teacher_id int NOT NULL AUTO_INCREMENT,
    teacher_name varchar(255) NOT NULL,
    teacher_mail varchar(255),
    teacher_password varchar(255),
    given_course_ids varchar(500),
    PRIMARY KEY (teacher_id)
);

CREATE TABLE project.students (
    student_id int NOT NULL AUTO_INCREMENT,
    student_name varchar(255) NOT NULL,
    student_mail varchar(255),
    student_password varchar(255),
    registered_course_ids varchar(500),
    PRIMARY KEY (student_id)
);

CREATE TABLE project.courses (
    course_id int NOT NULL AUTO_INCREMENT,
    course_name varchar(255) NOT NULL,
    course_description varchar(700),
    PRIMARY KEY (course_id)
);

CREATE TABLE project.posts (
    post_id int NOT NULL AUTO_INCREMENT,
    post_short_desc varchar(255) NOT NULL,
    post_long_desc varchar(700),
    post_date date,
    post_course_id int,
    post_student_id int,
    post_teacher_id int,
    PRIMARY KEY (post_id)
);

CREATE TABLE project.resources (
    resource_id int NOT NULL AUTO_INCREMENT,
    resource_short_desc varchar(255) NOT NULL,
    resource_long_desc varchar(700),
    resource_course_id int,
    resource_url varchar(255),
    PRIMARY KEY (resource_id)
);

CREATE TABLE project.homeworks (
    homework_id int NOT NULL AUTO_INCREMENT,
    homework_short_desc varchar(255) NOT NULL,
    homework_long_desc varchar(700),
    homework_course_id int,
    resource_file varchar(700),
    PRIMARY KEY (homework_id)
);
